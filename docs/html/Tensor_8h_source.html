272 inline float64 &Tensor::operator()(Args... indices)
273 {
274 const size_t idx[] = {indices...};
275 constexpr size_t isize = sizeof...(indices);
276 size_t nsize = isize;
277#ifdef CHECK_BOUNDS
278 if constexpr (isize > 1)
279 if (shape.size() < isize)
280 throw new std::range_error("Out of bound in Tensor () operetor");
281
282 if constexpr (isize == 0)
283 {
284 if (size == 0)
285 throw new std::range_error("Out of bound in Tensor () operetor");
286 }
287 else if constexpr (isize == 1)
288 {
289 if (size <= *idx)
290 throw new std::range_error("Out of bound in Tensor () operetor");
291 }
292 else if constexpr (isize == 2)
293 {
294 if (idx[0] >= *(shape.end() - 2) || idx[1] >= *(shape.end() - 1))
295 throw new std::range_error("Out of bound in Tensor () operetor");
296 }
297 else if constexpr (isize == 3)
298 {
299 if (idx[0] >= *(shape.end() - 3) || idx[1] >= *(shape.end() - 2) || idx[2] >= *(shape.end() - 1))
300 throw new std::range_error("Out of bound in Tensor () operetor");
301 }
302 else if constexpr (isize == 4)
303 {
304 if (idx[0] >= *(shape.end() - 4) || idx[1] >= *(shape.end() - 3) || idx[2] >= *(shape.end() - 2) || idx[3] >= *(shape.end() - 1))
305 throw new std::range_error("Out of bound in Tensor () operetor");
306 }
307 else
308 for (size_t i = 0; i < isize; i++)
309 {
310 if (idx[i] >= *(shape.end() - isize + i - 1))
311 throw new std::range_error("Out of bound in Tensor () operetor");
312 }
313#endif
314
315 if constexpr (isize == 0)
316 return this->b[0];
317 else if constexpr (isize == 1)
318 return this->b[*idx];
319 else if constexpr (isize == 2)
320 return this->b[idx[0] * shape.back() + idx[1]];
321 else if constexpr (isize == 3)
322 return this->b[(idx[0] * *(shape.end() - 2) + idx[1]) * shape.back() + idx[2]];
323 else if constexpr (isize == 4)
324 return this->b[((idx[0] * *(shape.end() - 3) + idx[1]) * *(shape.end() - 2) + idx[2]) * shape.back() + idx[3]];
325 else
326 {
327 size_t n = 0;
328 for (size_t i = 0; i < this->shape.size() - 1; i++)
329 n = (n + idx[i]) * this->shape[i + 1];
330
331 return this->b[n + idx[this->shape.size() - 1]];
332 }

407 template <size_t N>
408 inline DirectTensorView Tensor::sliceLastNDims(const std::vector<size_t> &index)
409 {
410 static_assert(N > 0);
411 if (index.size() + N > shape.size())
412 throw new std::range_error("Out of bound in Tensor sliceLastNDims()");
413
414 size_t new_shape[N], off = 0;
415 for (size_t i = 0; i < index.size(); i++)
416 {
417 off = (off + index[i]) * *(shape.end() - index.size() + i - N + 1);
418 if (index[i] >= *(shape.end() - index.size() + i - N))
419 throw new std::range_error("Out of bound in Tensor sliceLastNDims()");
420 }
421 for (size_t i = 0; i < N - 1; i++)
422 off *= *(shape.end() + i - N + 1);
423
424 for (size_t i = 0; i < N; i++)
425 new_shape[i] = *(shape.end() - N + i);
426
427 return DirectTensorView({new_shape, new_shape + N}, b + off);
428 }
