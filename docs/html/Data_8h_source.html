175 inline Tensor resizeImage(const Tensor& image, size_t new_height, size_t new_width)
176 {
177 if(new_height == 0 || new_width == 0)
178 throw std::runtime_error("New Image Size can't be of height or width of 0");
179
180 Tensor new_image({1, image.getShape()[1], new_height, new_width});
181
182 double width = image.getShape()[3];
183 double height = image.getShape()[2];
184
185 double w_ratio = width / (double) new_width;
186 double h_ratio = height / (double) new_height;
187
188 for (size_t i = 0; i < new_height; i++)
189 {
190 for (size_t j = 0; j < new_width; j++)
191 {
192 double x = w_ratio * (double) j;
193 double y = h_ratio * (double) i;
194 for (size_t k = 0; k < image.getShape()[1]; k++)
195 {
196 double x_1 = floor(x);
197 double x_2 = std::min(width - 1, ceil(x));
198 double y_1 = floor(y);
199 double y_2 = std::min(height- 1, ceil(y));
200
201 double x_weight = x - x_1;
202 double y_weight = y - y_1;
203
204 // Formula https://en.wikipedia.org/wiki/Bilinear_interpolation#On_the_unit_square
205 new_image(0UL, k, i, j) = image(0UL, k, (size_t)y_1, (size_t)x_1) * (1 - x_weight) * (1 - y_weight) +
206 image(0UL, k, (size_t)y_1, (size_t)x_2) * x_weight * (1 - y_weight) +
207 image(0UL, k, (size_t)y_2, (size_t)x_1) * y_weight * (1 - x_weight) +
208 image(0UL, k, (size_t)y_2, (size_t)x_2) * x_weight * y_weight;
209 }
210 }
211 }
212
213 return new_image;
214 }
